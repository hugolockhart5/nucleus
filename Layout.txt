import React from "react";
import { Link, useLocation } from "react-router-dom";
import { createPageUrl } from "@/utils";
import { base44 } from "@/api/base44Client";
import { useQuery } from "@tanstack/react-query";
import { 
  Zap, 
  Calendar, 
  History, 
  UserCircle, 
  LogOut,
  LayoutDashboard,
  Users,
  FileText,
  DollarSign
} from "lucide-react";
import {
  Sidebar,
  SidebarContent,
  SidebarGroup,
  SidebarGroupContent,
  SidebarMenu,
  SidebarMenuButton,
  SidebarMenuItem,
  SidebarHeader,
  SidebarFooter,
  SidebarProvider,
  SidebarTrigger,
} from "@/components/ui/sidebar";
import { Button } from "@/components/ui/button";

export default function Layout({ children, currentPageName }) {
  const location = useLocation();
  const [user, setUser] = React.useState(null);
  const [isExpert, setIsExpert] = React.useState(false);
  const [isAdmin, setIsAdmin] = React.useState(false);

  React.useEffect(() => {
    const loadUser = async () => {
      try {
        const currentUser = await base44.auth.me();
        setUser(currentUser);
        setIsAdmin(currentUser.role === 'admin');
        
        const experts = await base44.entities.Expert.filter({ user_id: currentUser.id });
        setIsExpert(experts.length > 0);
      } catch (error) {
        console.log("No user logged in");
      }
    };
    loadUser();
  }, []);

  const buyerNavigation = [
    { title: "Find Expert", url: createPageUrl("Home"), icon: Zap },
    { title: "My Sessions", url: createPageUrl("MySessions"), icon: Calendar },
  ];

  const expertNavigation = [
    { title: "Expert Dashboard", url: createPageUrl("ExpertDashboard"), icon: LayoutDashboard },
    { title: "Availability", url: createPageUrl("ExpertAvailability"), icon: Calendar },
    { title: "Earnings", url: createPageUrl("ExpertEarnings"), icon: DollarSign },
  ];

  const adminNavigation = [
    { title: "Vetting Queue", url: createPageUrl("AdminVetting"), icon: Users },
    { title: "All Sessions", url: createPageUrl("AdminSessions"), icon: FileText },
  ];

  const handleLogout = async () => {
    await base44.auth.logout();
  };

  // Public pages don't need sidebar
  const publicPages = ["Home", "ApplyExpert"];
  const isPublicPage = publicPages.includes(currentPageName) || !user;

  if (isPublicPage) {
    return (
      <div className="min-h-screen bg-gradient-to-br from-slate-50 to-blue-50">
        <style>
          {`
            :root {
              --primary-600: #2563eb;
              --primary-700: #1d4ed8;
              --accent-500: #f59e0b;
              --accent-600: #d97706;
            }
          `}
        </style>
        {children}
      </div>
    );
  }

  return (
    <SidebarProvider>
      <style>
        {`
          :root {
            --primary-600: #2563eb;
            --primary-700: #1d4ed8;
            --accent-500: #f59e0b;
            --accent-600: #d97706;
          }
        `}
      </style>
      <div className="min-h-screen flex w-full bg-slate-50">
        <Sidebar className="border-r border-slate-200">
          <SidebarHeader className="border-b border-slate-200 p-6">
            <Link to={createPageUrl("Home")} className="flex items-center gap-3">
              <div className="w-10 h-10 bg-gradient-to-br from-blue-600 to-blue-700 rounded-xl flex items-center justify-center shadow-lg">
                <Zap className="w-6 h-6 text-white" />
              </div>
              <div>
                <h2 className="font-bold text-xl text-slate-900">Nucleus</h2>
                <p className="text-xs text-slate-500">Decisions, clarified</p>
              </div>
            </Link>
          </SidebarHeader>
          
          <SidebarContent className="p-2">
            {/* Buyer navigation - always show */}
            <SidebarGroup>
              <SidebarGroupContent>
                <SidebarMenu>
                  {buyerNavigation.map((item) => (
                    <SidebarMenuItem key={item.title}>
                      <SidebarMenuButton 
                        asChild 
                        className={`hover:bg-blue-50 hover:text-blue-700 transition-colors mb-1 rounded-lg ${
                          location.pathname === item.url ? 'bg-blue-50 text-blue-700 font-medium' : ''
                        }`}
                      >
                        <Link to={item.url} className="flex items-center gap-3 px-3 py-2">
                          <item.icon className="w-4 h-4" />
                          <span>{item.title}</span>
                        </Link>
                      </SidebarMenuButton>
                    </SidebarMenuItem>
                  ))}
                </SidebarMenu>
              </SidebarGroupContent>
            </SidebarGroup>

            {/* Expert navigation - show if user is expert */}
            {isExpert && (
              <SidebarGroup>
                <div className="px-3 py-2 text-xs font-semibold text-slate-500 uppercase tracking-wider">
                  Expert Panel
                </div>
                <SidebarGroupContent>
                  <SidebarMenu>
                    {expertNavigation.map((item) => (
                      <SidebarMenuItem key={item.title}>
                        <SidebarMenuButton 
                          asChild 
                          className={`hover:bg-amber-50 hover:text-amber-700 transition-colors mb-1 rounded-lg ${
                            location.pathname === item.url ? 'bg-amber-50 text-amber-700 font-medium' : ''
                          }`}
                        >
                          <Link to={item.url} className="flex items-center gap-3 px-3 py-2">
                            <item.icon className="w-4 h-4" />
                            <span>{item.title}</span>
                          </Link>
                        </SidebarMenuButton>
                      </SidebarMenuItem>
                    ))}
                  </SidebarMenu>
                </SidebarGroupContent>
              </SidebarGroup>
            )}

            {/* Admin navigation - show if admin */}
            {isAdmin && (
              <SidebarGroup>
                <div className="px-3 py-2 text-xs font-semibold text-slate-500 uppercase tracking-wider">
                  Admin
                </div>
                <SidebarGroupContent>
                  <SidebarMenu>
                    {adminNavigation.map((item) => (
                      <SidebarMenuItem key={item.title}>
                        <SidebarMenuButton 
                          asChild 
                          className={`hover:bg-red-50 hover:text-red-700 transition-colors mb-1 rounded-lg ${
                            location.pathname === item.url ? 'bg-red-50 text-red-700 font-medium' : ''
                          }`}
                        >
                          <Link to={item.url} className="flex items-center gap-3 px-3 py-2">
                            <item.icon className="w-4 h-4" />
                            <span>{item.title}</span>
                          </Link>
                        </SidebarMenuButton>
                      </SidebarMenuItem>
                    ))}
                  </SidebarMenu>
                </SidebarGroupContent>
              </SidebarGroup>
            )}

            {/* Apply to be expert - show if not expert */}
            {!isExpert && (
              <SidebarGroup className="mt-4">
                <SidebarGroupContent>
                  <Link to={createPageUrl("ApplyExpert")} className="block">
                    <div className="mx-3 p-4 bg-gradient-to-br from-amber-50 to-orange-50 border border-amber-200 rounded-lg hover:shadow-md transition-shadow">
                      <div className="flex items-center gap-2 mb-2">
                        <Users className="w-4 h-4 text-amber-700" />
                        <span className="font-semibold text-sm text-amber-900">Become an Expert</span>
                      </div>
                      <p className="text-xs text-amber-700">Share your expertise and earn</p>
                    </div>
                  </Link>
                </SidebarGroupContent>
              </SidebarGroup>
            )}
          </SidebarContent>

          <SidebarFooter className="border-t border-slate-200 p-4">
            {user && (
              <div className="space-y-3">
                <div className="flex items-center gap-3 px-2">
                  <div className="w-8 h-8 bg-blue-100 rounded-full flex items-center justify-center">
                    <UserCircle className="w-5 h-5 text-blue-600" />
                  </div>
                  <div className="flex-1 min-w-0">
                    <p className="font-medium text-sm text-slate-900 truncate">{user.full_name || user.email}</p>
                    <p className="text-xs text-slate-500 truncate">{user.email}</p>
                  </div>
                </div>
                <Button
                  variant="ghost"
                  size="sm"
                  onClick={handleLogout}
                  className="w-full justify-start text-slate-600 hover:text-slate-900"
                >
                  <LogOut className="w-4 h-4 mr-2" />
                  Log out
                </Button>
              </div>
            )}
          </SidebarFooter>
        </Sidebar>

        <main className="flex-1 flex flex-col">
          <header className="bg-white border-b border-slate-200 px-6 py-4 lg:hidden">
            <div className="flex items-center gap-4">
              <SidebarTrigger className="hover:bg-slate-100 p-2 rounded-lg" />
              <h1 className="text-xl font-bold">Nucleus</h1>
            </div>
          </header>

          <div className="flex-1 overflow-auto bg-slate-50">
            {children}
          </div>
        </main>
      </div>
    </SidebarProvider>
  );
}
